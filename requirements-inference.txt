# PyTorch requires a special wheel index (download.pytorch.org/whl/cu124) that
# cannot be expressed inside a requirements file without also affecting other
# packages.  It is installed as a separate explicit step in the Dockerfile
# builder stage before this file is processed:
#
#   RUN pip install torch==2.6.0+cu124 torchvision==0.21.0+cu124 \
#       torchaudio==2.6.0+cu124 \
#       --extra-index-url https://download.pytorch.org/whl/cu124
#
# If you install this file standalone (outside Docker), install torch first:
#   pip install torch==2.6.0+cu124 --extra-index-url https://download.pytorch.org/whl/cu124
#   pip install -r requirements-inference.txt
#
# For CPU-only environments:
#   pip install torch==2.6.0  # from PyPI
#   pip install -r requirements-inference.txt
#
# [MEDIUM]: This file is now wired into the Dockerfile builder stage.
#   docker/Dockerfile builder stage:
#     COPY requirements-inference.txt /tmp/requirements.txt
#     RUN pip install --no-cache-dir -r /tmp/requirements.txt
#   Previously the Dockerfile still used requirements.txt, making this file
#   dead configuration. See docker/Dockerfile for the corrected COPY line.
#
# [HIGH]: Added matplotlib==3.10.6.
#   shap==0.48.0 is listed below and used for inference-time explanations.
#   SHAP's plot functions (shap.plots.*, force_plot, waterfall_plot) have a
#   hard import-time dependency on matplotlib. Without it the inference
#   container raises ImportError on any SHAP visualisation call.
#   If SHAP is used only for shap_values() computation (no plots), matplotlib
#   is still the safer default because shap conditionally imports it at the
#   top of several submodules.
#
